TOURNOI 

    1 - Create : 
        POST : /api.tournoi/create
        Content-type = JSON
        body :
            {
                'NomTournoi' : string required, 
                'DateDebutTournoi' : string (yyyy-mm-dd) required, 
                'DateFinTournoi' : string (yyyy-mm-dd) required , 
                'LieuTournoi' : string required 
            }

        
        response : 
            - success : 
                Content-type : JSON

                body : 
                    {
                        'NomTournoi' : string , 
                        'DateDebutTournoi' : string ,
                        'DateFinTournoi' : string ,
                        'LieuTournoi' : string , 
                        'Equipes': [
                            {
                                'NomEquipe' : string,
                                'RefEquipe' : int
                            } , 
                            // ... equipe hafa (liste)
                        ]
                    }

            - failure 

        2 - Read(par id) : 

                GET : /api.tournoiNid=${RefTournoi}

                response : *

                    - success : 
                                    
                        Content-type : JSON

                        Body : 
                            {
                                'RefTournoi' : int
                                'NomTournoi' : string , 
                                'DateDebutTournoi' : string ,
                                'DateFinTournoi' : string ,
                                'LieuTournoi' : string ,  
                                'Equipes': [
                                                {
                                                    'NomEquipe' : string,
                                                '   RefEquipe' : int
                                                 }                        , 
                                    // ... equipe hafa (liste)
                                            ]
                    }

            - failure 

        3 - Read all (liste de tous les tournois) : 

                GET :  /api.tournoi/all

                response : 
                    - success : 
                            Content-type : JSON

                            [
                                {   
                                    'RefTournoi' : int, 
                                    'NomTournoi' : string , 
                                    'DateDebutTournoi' : string ,
                                    'DateFinTournoi' : string ,
                                    'LieuTournoi' : string , 
                                }
                            ]
